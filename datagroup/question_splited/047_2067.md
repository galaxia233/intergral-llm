$$
\begin{array}{l} \int P (x) \cos a x d x = \frac {\sin a x}{a} \left[ P (x) - \frac {P ^ {\prime \prime} (x)}{a ^ {2}} + \frac {P ^ {(4)} (x)}{a ^ {4}} - \dots \right] + \frac {\cos a x}{a ^ {2}} \left[ P ^ {\prime} (x) - \frac {P ^ {\prime \prime} (x)}{a ^ {2}} + \frac {P ^ {(5)} (x)}{a ^ {4}} - \dots \right] + C, \\ \int P (x) \sin a x d x = - \frac {\cos a x}{a} \left[ P (x) - \frac {P ^ {\prime \prime} (x)}{a ^ {2}} + \frac {P ^ {(4)} (x)}{a ^ {4}} - \dots \right] + \frac {\sin a x}{a ^ {2}} \left[ P ^ {\prime} (x) - \frac {P ^ {\prime \prime} (x)}{a ^ {2}} + \frac {P ^ {(5)} (x)}{a ^ {4}} - \dots \right] + C. \\ \end{array}
$$

证 $\int P(x)\cos ax\mathrm{d}x = \frac{1}{a}\int P(x)\mathrm{d}(\sin ax) = \frac{1}{a} P(x)\sin ax - \frac{1}{a}\int P'(x)\sin ax\mathrm{d}x$

$$
\begin{array}{l} = \frac {1}{a} P (x) \sin a x + \frac {1}{a ^ {2}} \int P ^ {\prime} (x) d (\cos a x) = \frac {1}{a} P (x) \sin a x + \frac {1}{a ^ {2}} P ^ {\prime} (x) \cos a x - \frac {1}{a ^ {2}} \int P ^ {\prime \prime} (x) \cos a x d x \\ = \frac {1}{a} P (x) \sin a x + \frac {1}{a ^ {2}} P ^ {\prime} (x) \cos a x - \frac {1}{a ^ {3}} P ^ {\prime \prime} (x) \sin a x - \frac {1}{a ^ {4}} P ^ {\prime \prime} (x) \cos a x + \frac {1}{a ^ {4}} \int P ^ {(4)} (x) \cos a x d x \\ = \dots = \frac {\sin a x}{a} \left[ P (x) - \frac {P ^ {\prime \prime} (x)}{a ^ {2}} + \frac {P ^ {(4)} (x)}{a ^ {4}} - \dots \right] + \frac {\cos a x}{a ^ {2}} \left[ P ^ {\prime} (x) - \frac {P ^ {\prime \prime} (x)}{a ^ {2}} + \frac {P ^ {(5)} (x)}{a ^ {4}} - \dots \right] + C. \\ \int P (x) \sin a x d x = - \frac {1}{a} \int P (x) d (\cos a x) = - \frac {1}{a} P (x) \cos a x + \frac {1}{a} \int P ^ {\prime} (x) \cos a x d x \\ = - \frac {1}{a} P (x) \cos a x + \frac {1}{a ^ {2}} \int P ^ {\prime} (x) d (\sin a x) = - \frac {1}{a} P (x) \cos a x + \frac {1}{a ^ {2}} P ^ {\prime} (x) \sin a x - \frac {1}{a ^ {2}} \int P ^ {\prime \prime} (x) \sin a x d x \\ = - \frac {1}{a} P (x) \cos a x + \frac {1}{a ^ {2}} P ^ {\prime} (x) \sin a x + \frac {1}{a ^ {3}} P ^ {\prime \prime} (x) \cos a x - \frac {1}{a ^ {4}} P ^ {\prime \prime} (x) \cos a x + \frac {1}{a ^ {4}} \int P ^ {(4)} (x) \cos a x d x \\ = \dots = - \frac {\cos a x}{a} \left[ P (x) - \frac {P ^ {\prime \prime} (x)}{a ^ {2}} + \frac {P ^ {(4)} (x)}{a ^ {4}} - \dots \right] + \frac {\sin a x}{a ^ {2}} \left[ P ^ {\prime} (x) - \frac {P ^ {\prime \prime} (x)}{a ^ {2}} + \frac {P ^ {(5)} (x)}{a ^ {4}} - \dots \right] + C. \\ \end{array}
$$
