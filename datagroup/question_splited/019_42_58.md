$$
\begin{array}{l} \int_ {0} ^ {\pi} \sin x y d y = - \frac {1}{x} \cos x y \Bigg ] _ {0} ^ {\pi} = - \frac {1}{x} (\cos \pi x - \cos 0) = - \frac {1}{x} (\cos \pi x - 1). \\ \frac {d}{d x} \left[ - \frac {1}{x} (\cos \pi x - 1) \right] = - \frac {x (- \pi \sin \pi x) - (\cos \pi x - 1)}{x ^ {2}} = \frac {\cos \pi x + \pi x \sin \pi x - 1}{x ^ {2}} \\ \partial f / \partial x = y \cos x y \quad \text {a n d} \quad \int_ {0} ^ {\pi} y \cos x y d y \quad \text {c a n b e i n t e g r a t e d b y p a r t s :} \end{array}
$$
