$$
\begin{array}{l} \mathbf {F} \times \mathbf {G} = (g \eta - h \psi , h \phi - f \eta , f \psi - g \phi). \quad \text {H e n c e}, \quad \operatorname {d i v} (\mathbf {F} \times \mathbf {G}) = (g \eta - h \psi) _ {x} + (h \phi - f \eta) _ {y} + (f \psi - g \phi) _ {z} = \\ g \eta_ {x} + g _ {x} \eta - h \psi_ {x} - h _ {x} \psi + h \phi_ {y} + h _ {y} \phi - f \eta_ {y} - f _ {y} \eta + f \psi_ {z} + f _ {z} \psi - g \phi_ {z} - g _ {z} \phi = \quad \phi (h _ {y} - g _ {z}) + \psi (f _ {z} - h _ {x}) + \eta (g _ {x} - \\ f _ {y}) + f (\psi_ {z} - \eta_ {y}) + g (\eta_ {x} - \phi_ {z}) + h (\phi_ {y} - \psi_ {x}) = \mathbf {G} \cdot \operatorname {c u r l} \mathbf {F} - \mathbf {F} \cdot \operatorname {c u r l} \mathbf {G}. \end{array}
