$$
\begin{array}{l} b _ {n} = \frac {2}{\pi} \int_ {0} ^ {\pi} \arcsin (\sin x) \sin n x d x = \frac {2}{\pi} \left[ \int_ {0} ^ {\frac {\pi}{2}} x \sin n x d x + \int_ {\frac {\pi}{2}} ^ {\pi} (\pi - x) \sin n x d x \right] \\ = \frac {2}{\pi} \left[ \right.\left(- \frac {x}{n} \cos n x + \frac {1}{n ^ {2}} \sin n x\right)\left| _ {0} ^ {\frac {\pi}{2}} - \frac {\pi}{n} \cos n x \right| _ {\frac {\pi}{2}} ^ {n} + \left(\frac {x}{n} \cos n x - \frac {1}{n ^ {2}} \sin n x\right)\left| \right. _ {\frac {\pi}{2}} ^ {n} \left. \right] = \frac {4}{n ^ {2} \pi} \sin \frac {n \pi}{2} \\ = \left\{ \begin{array}{l l} 0, & n = 2 k (k = 1, 2, 3, \dots), \\ (- 1) ^ {k} \frac {4}{\pi (2 k + 1) ^ {2}}, & n = 2 k + 1 (k = 0, 1, 2, \dots), \end{array} \right. \\ \end{array}
